## Clone Graph

Possible to make shorter and faster.

```python
"""
# Definition for a Node.
class Node:
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []
"""

class Solution:
    def cloneGraph(self, node: 'Node') -> 'Node':
        DICT_NODES = dict()
        def dfs_new_node(node):
            if node.val in DICT_NODES:
                return DICT_NODES[node.val]
            new_node = Node(node.val)
            DICT_NODES.update({node.val: new_node})
            if node.neighbors:
                for ele in node.neighbors:
                    DICT_NODES.update({ele.val: dfs_new_node(ele)})
                    new_node.neighbors.append(DICT_NODES[ele.val])
            return new_node

        return dfs_new_node(node) if node else None
```
