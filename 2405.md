```python
class Solution:
    def partitionString(self, s: str) -> int:
        res = 1
        hash_num = {'a': 2, 'b': 3, 'c': 5, 'd': 7, 'e': 11, 'f': 13, 'g': 17, 'h': 19, 'i': 23, 'j': 29, 'k': 31, 'l': 37, 'm': 41, 'n': 43, 'o': 47, 'p': 53, 'q': 59, 'r': 61, 's': 67, 't': 71, 'u': 73, 'v': 79, 'w': 83, 'x': 89, 'y': 97, 'z': 101}
        cur = 1
        for ele in s:
            if cur % hash_num[ele] == 0:
                res += 1
                cur = hash_num[ele]
            else:
                cur *= hash_num[ele]
        return res
```
