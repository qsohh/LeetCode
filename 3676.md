## Count Bowl Subarrays

```cpp
class Solution {
public:
    long long bowlSubarrays(vector<int>& nums) {
        std::vector<int> st;
        long long res=0;
        for (int n: nums) {
            while ((!st.empty()) && st.back() < n) {
                st.pop_back();
                if (!st.empty()) {res++;}
            }
            st.push_back(n);
        }
        return res;
    }
};
```
