class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        n_raw, n_line = len(grid), len(grid[0])
        flag_visited = [[ele for ele in raw] for raw in grid]
        res = 0
        def dfs(x, y):
            if (x < 0) or (x > (n_raw-1)) or (y < 0) or (y > (n_line-1)):
                return None
            if flag_visited[x][y]:
                return None
            flag_visited[x][y] = 1
            dfs(x+1, y)
            dfs(x, y+1)
            dfs(x-1, y)
            dfs(x, y-1)

        for x in range(n_raw):
            if not flag_visited[x][0]:
                dfs(x, 0)
            if not flag_visited[x][-1]:
                dfs(x, n_line-1)
        for y in range(n_line):
            if not flag_visited[0][y]:
                dfs(0, y)
            if not flag_visited[-1][y]:
                dfs(n_raw-1, y)

        print(flag_visited)
        for x in range(1, n_raw-1):
            for y in range(1, n_line-1):
                if not flag_visited[x][y]:
                    res += 1
                    dfs(x, y)

        return res
