## Number of Enclaves

Same as **1254**.

```python
class Solution:
    def numEnclaves(self, grid: List[List[int]]) -> int:
        n_raw, n_column = len(grid), len(grid[0])
        flag_visited = [[ele for ele in raw] for raw in grid]
        def dfs(x, y) -> int:
            if (x < 0) or (x > (n_raw-1)) or (y < 0) or (y > (n_column-1)):
                return 0
            if not flag_visited[x][y]:
                return 0
            flag_visited[x][y] = 0
            return dfs(x+1, y) + dfs(x, y+1) + dfs(x-1, y) + dfs(x, y-1) + 1

        for x in range(n_raw):
            if flag_visited[x][0]:
                dfs(x, 0)
            if flag_visited[x][-1]:
                dfs(x, n_column-1)
        for y in range(n_column):
            if flag_visited[0][y]:
                dfs(0, y)
            if flag_visited[-1][y]:
                dfs(n_raw-1, y)

        res = 0
        for x in range(1, n_raw-1):
            for y in range(1, n_column-1):
                if flag_visited[x][y]:
                    res += dfs(x, y)
        return res
```
