## 2348. Number of Zero-Filled Subarrays

但是实际上每次直接加一个cur上去，可能比现在这样要快一些。

```cpp
class Solution {
public:
    long long zeroFilledSubarray(vector<int>& nums) {
        long long int res = 0;
        long long int cur = 0;
        bool last_zero = false;
        for (auto& n: nums) {
            if (n == 0 && last_zero) { cur++; }
            else if (n == 0 && !last_zero) { cur = 1; last_zero = true; }
            else if (n != 0 && last_zero) { res += cur*(cur+1)/2; last_zero = false; }
        }
        return res + (last_zero ? cur*(cur+1)/2 : 0);
    }
};
```
