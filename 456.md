## 132 Pattern

Approach 4 的 stack 法挺好的

```python
class Solution:
    def find132pattern(self, nums: List[int]) -> bool:
        cur_min, mins_nums, nb = nums[0], [], len(nums)
        stack = []
        for i, ele in enumerate(nums):
            if ele < cur_min:
                cur_min = ele
            mins_nums.append(cur_min)
        
        for i, ele in enumerate(nums[::-1]):
            if ele > mins_nums[nb-i-1]:
                if not stack:
                    stack.append(ele)
                    continue
                if ele < stack[-1]:
                    stack.append(ele)
                    continue
                if ele == stack[-1]:
                    continue
                while stack:
                    if stack[-1] > mins_nums[nb-i-1]:
                        return True
                    else:
                        stack.pop()
                stack.append(ele)
        return False
```
