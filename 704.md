```python
class Solution:
    def search(self, nums: List[int], target: int) -> int:
        p1, p2 = 0, len(nums)-1
        while p2 >= p1:
            p = (p1 + p2)//2
            if nums[p] == target:
                return p
            elif nums[p] > target:
                p2 = p - 1
            elif nums[p] < target:
                p1 = p + 1
        return -1
```
