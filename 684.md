## Redundant Connection

DSU (Disjoint Set Union)要在find的时候进行压缩，而不是在unite的时候压。

```cpp
class Solution {
private:
    std::vector<int> _parent;
    std::vector<int> _rank;

    int _find(const int& idx) {
        if (_parent[idx] == idx) {return idx;}
        _parent[idx] = _find(_parent[idx]);
        return _parent[idx];
    }

    bool _unite(const int& a, const int& b) {
        int fa = _find(a), fb = _find(b);
        if (fa == fb) {return true;}
        if (_rank[a] > _rank[b]) {_parent[fb] = fa;}
        else {_parent[fa] = fb;}
        if (_rank[a] == _rank[b]) {_rank[a]++;}
        return false;
    }

public:
    vector<int> findRedundantConnection(const vector<vector<int>>& edges) {
        int n = edges.size();
        _parent.resize(n+1);
        std::iota(_parent.begin(), _parent.end(), 0);
        _rank.assign(n+1, 1);
        for (const auto& e: edges) {
            if (_unite(e[0], e[1])) {return e;}
        }
        return std::vector<int>{0, 0}; // never reached
    }
};
```
