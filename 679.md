## 24 Game

需要注意的一点是，$((\cdot, \cdot), \cdot), \cdot$ 与 $\cdot, (\cdot, (\cdot, \cdot))$是不同的。

```python
CORD = [[0, 1, 2, 3], [0, 1, 3, 2], [0, 2, 1, 3], [0, 2, 3, 1], [0, 3, 1, 2], [0, 3, 2, 1], [1, 0, 2, 3], [1, 0, 3, 2], [1, 2, 0, 3], [1, 2, 3, 0], [1, 3, 0, 2], [1, 3, 2, 0], [2, 0, 1, 3], [2, 0, 3, 1], [2, 1, 0, 3], [2, 1, 3, 0], [2, 3, 0, 1], [2, 3, 1, 0], [3, 0, 1, 2], [3, 0, 2, 1], [3, 1, 0, 2], [3, 1, 2, 0], [3, 2, 0, 1], [3, 2, 1, 0]]
OPER = [lambda a, b: a+b, lambda a, b: a-b, lambda a, b: a*b, lambda a, b: a/b if b != 0 else -pi]
EPS = 1e-4

class Solution:
    def judgePoint24(self, cards: List[int]) -> bool:
        def h1(c):
            for i in OPER:
                for j in OPER:
                    for k in OPER:
                        if 24-EPS < k(i(cards[c[0]], cards[c[1]]), j(cards[c[2]], cards[c[3]])) < 24+EPS:
                            return True
            return False
        
        def h2(c):
            for i in OPER:
                for j in OPER:
                    for k in OPER:
                        if 24-EPS < k(j(i(cards[c[0]], cards[c[1]]), cards[c[2]]), cards[c[3]]) < 24+EPS:
                            return True
            return False
        
        def h3(c):
            for i in OPER:
                for j in OPER:
                    for k in OPER:
                        if 24-EPS < i(cards[c[0]], j(cards[c[1]], k(cards[c[2]], cards[c[3]]))) < 24+EPS:
                            return True
            return False

        for c in CORD:
            if h1(c) or h2(c) or h3(c):
                return True
        return False
```
