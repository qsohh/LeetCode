## Unique Binary Search Trees II

分治法搞定 ：

```python
class Solution:
    def generateTrees(self, n: int) -> List[Optional[TreeNode]]:
        @lru_cache(None)
        def helper(ind_left, ind_right):
            if ind_left > ind_right:
                return [None]
            elif ind_left == ind_right:
                return [TreeNode(ind_left)]
            else:
                res = []
                for i in range(ind_left, ind_right+1):
                    for ele_left, ele_right in product(helper(ind_left, i-1), helper(i+1, ind_right)):
                        res.append(TreeNode(i, ele_left, ele_right))
                return res
        return helper(1, n)
```
